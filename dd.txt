clt<-function(n){

 parm_distribution<-rep(0,1000)

 #從母群體抽出樣本大小為n的樣本，並計算其統計量，重複此過程以產生1000個統計量
 for (i in 1:1000){
  parm_distribution[i]<-var(rpois(n,1))
 }
 name<-paste("sample size=",n,sep="")

 #畫出抽樣數為n的抽樣分布（統計量的機率分布）直方圖
 hist(parm_distribution,col="green",main=name)
}

#畫出抽樣數為2、4、8...、4096的抽樣分布
par(mfrow=c(3,4))
